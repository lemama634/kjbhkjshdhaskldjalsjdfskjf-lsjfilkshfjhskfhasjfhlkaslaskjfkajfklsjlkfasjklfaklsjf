local function ESPYLN_fake_script() -- TextButton_54.LocalScript 
	local script = Instance.new('LocalScript', TextButton_54)

	script.Parent.MouseButton1Click:Connect(function()
		local player = game.Players.LocalPlayer
		local char = player.Character
		local tool
	
		-- Find the tool with SyncAPI
		for i, v in player:GetDescendants() do
			if v.Name == "SyncAPI" then
				tool = v.Parent
			end
		end
		for i, v in game.ReplicatedStorage:GetDescendants() do
			if v.Name == "SyncAPI" then
				tool = v.Parent
			end
		end
	
		local remote = tool.SyncAPI.ServerEndpoint
		function _(args)
			remote:InvokeServer(unpack(args))
		end
	
		function CreateBigSquare(cf)
			local args = {
				[1] = "CreatePart",
				[2] = "Normal",
				[3] = cf,
				[4] = workspace
			}
			_(args)
		end
	
		function ResizeBigSquare(part)
			local args = {
				[1] = "SyncResize",
				[2] = {
					[1] = {
						["Part"] = part,
						["CFrame"] = part.CFrame,
						["Size"] = Vector3.new(500, 500, 500) -- Super big square!
					}
				}
			}
			_(args)
		end
	
		function SetAnchor(part, anchored)
			local args = {
				[1] = "SyncAnchor",
				[2] = {
					[1] = {
						["Part"] = part,
						["Anchored"] = anchored
					}
				}
			}
			_(args)
		end
	
		function ApplyTexture(part)
			local faces = {"Front", "Back", "Left", "Right", "Top", "Bottom"}
			for _, face in pairs(faces) do
				local args = {
					[1] = "CreateTexture",
					[2] = {
						["Part"] = part,
						["Face"] = face,
						["TextureId"] = "rbxassetid://77869853802996"
					}
				}
				_(args)
			end
		end
	
		hrpcf = player.Character.HumanoidRootPart.CFrame
		for i = 1, 10 do -- Limit the loop to 10 squares to prevent lag
			wait(0.5)
			local xloc = hrpcf.x + math.random(-600, 600)
			local zloc = hrpcf.z + math.random(-600, 600)
			local yloc = hrpcf.y + 400 + math.random(0, 400)
	
			spawn(function()
				CreateBigSquare(CFrame.new(xloc, yloc, zloc))
				for _, v in pairs(workspace:GetDescendants()) do
					if v:IsA("Part") and v.Position == Vector3.new(xloc, yloc, zloc) then
						SetAnchor(v, false) -- ✅ Set it unanchored BEFORE resizing
						ResizeBigSquare(v)
						ApplyTexture(v) -- ✅ Ensure texture on all sides
					end
				end
			end)
		end
	end)
	
end
